<!-- ShopSupport Pro Chat Widget -->
<script>
  window.SHOPSUPPORT_CONFIG = {
    serverUrl: 'REPLACE_WITH_YOUR_SERVER_URL', // Your ShopSupport Pro URL
    {% if customer %}
    customer: {
      id: {{ customer.id | json }},
      email: {{ customer.email | json }},
      name: {{ customer.first_name | append: ' ' | append: customer.last_name | json }},
      phone: {{ customer.phone | json }}
    }
    {% endif %}
  };
</script>
<script src="REPLACE_WITH_YOUR_SERVER_URL/widget.js" async></script>
